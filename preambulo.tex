% Resumen del capítulo


\newsavebox{\myboxb}
\newenvironment{Resumen}
{%
	\vspace*{-2.0cm}
	\fboxsep = 0pt
	\fboxrule = 0pt
	\begin{lrbox}{\myboxb}%
		\begin{minipage}{.85\textwidth}\itshape\parskip=2ex\parindent=2em
			}
			{%
		\end{minipage}
	\end{lrbox}%
	\begin{flushright}
		\usebox{\myboxb}%
	\end{flushright}
	\vspace{0.5cm}
}
